                    # Numba/ njit

from numba import jit
import numpy as np

# @vectorize

x = np.arange(100).reshape(10, 10)

@njit
def go_fast(a): # Function is compiled to machine code when called the first time
    trace = 0.0
    for i in range(a.shape[0]):   
        trace += np.tanh(a[i, i])
    return a + trace             

print(go_fast(x))
